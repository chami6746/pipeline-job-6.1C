pipeline {
    agent any
    
    tools {
        maven 'Maven 3.8.4' // Ensure this matches the Maven installation name configured in Jenkins
    }
    
    stages {
        stage('Build') {
            steps {
                script {
                    echo 'Building the code...'
                    // Tool: Maven
                    sh 'mvn clean install'
                }
            }
        }
        
        stage('Unit and Integration Tests') {
            steps {
                script {
                    echo 'Running unit and integration tests...'
                    // Tool: JUnit
                    sh 'mvn test'
                    sh 'mvn verify'
                }
            }
        }
        
        stage('Code Analysis') {
            steps {
                script {
                    echo 'Analyzing code...'
                    // Tool: SonarQube
                    sh 'mvn sonar:sonar'
                }
            }
        }
        
        stage('Security Scan') {
            steps {
                script {
                    echo 'Performing security scan...'
                    // Tool: OWASP Dependency-Check
                    sh 'dependency-check.sh --project my-project'
                }
            }
        }
        
        stage('Deploy to Staging') {
            steps {
                script {
                    echo 'Deploying to staging...'
                    // Tool: AWS CLI
                    sh 'aws deploy push --application-name my-app --s3-location s3://my-bucket/my-app.zip'
                }
            }
        }
        
        stage('Integration Tests on Staging') {
            steps {
                script {
                    echo 'Running integration tests on staging...'
                    // Tool: JUnit or other integration testing tools
                    sh 'mvn verify -Pstaging'
                }
            }
        }
        
        stage('Deploy to Production') {
            steps {
                script {
                    echo 'Deploying to production...'
                    // Tool: AWS CLI
                    sh 'aws deploy push --application-name my-app --s3-location s3://my-bucket/my-app.zip'
                }
            }
        }
    }

    post {
        success {
            echo 'Pipeline succeeded!'
            // Send email notification on success
            emailext (
                to: 'hasini.chameeliya@gmail.com',
                subject: 'Jenkins Pipeline Succeeded',
                body: 'The Jenkins pipeline has succeeded. Check the build details for more information.',
                attachLog: true
            )
        }
        failure {
            echo 'Pipeline failed.'
            // Send email notification on failure
            emailext (
                to: 'hasini.chameeliya@gmail.com',
                subject: 'Jenkins Pipeline Failed',
                body: 'The Jenkins pipeline has failed. Check the logs for more details.',
                attachLog: true
            )
        }
        unstable {
            echo 'Pipeline is unstable.'
            // Send email notification on unstable state
            emailext (
                to: 'hasini.chameeliya@gmail.com',
                subject: 'Jenkins Pipeline Unstable',
                body: 'The Jenkins pipeline is unstable. Check the logs for more details.',
                attachLog: true
            )
        }
    }
}
